38 bool ft_co_calc_pot_hits(t_cone co, t_ray ray, float potential_hits[3])
39 {
40  t_vec3 ray_apex;
41  t_equation eq;
42  float dot_product_ray_axis;
43  float dot_product_apex_axis;
44  float cos_2_angle;
45 
46  eq = (t_equation){0};
47  ray_apex = ft_vec3_sub(ray.origin, co.apex);
48  dot_product_ray_axis = ft_vec3_dot(ray.direction, co.axis);
49  dot_product_apex_axis = ft_vec3_dot(ray_apex, co.axis);
50  cos_2_angle = cos(co.angle) * cos(co.angle);
51  eq.a = dot_product_ray_axis * dot_product_ray_axis - cos_2_angle;
52  eq.b = 2.0 * (dot_product_ray_axis * dot_product_apex_axis - \
53  ft_vec3_dot(ray.direction, ray_apex) * cos_2_angle);
54  eq.c = dot_product_apex_axis * dot_product_apex_axis \
55  - ft_vec3_dot(ray_apex, ray_apex) * cos_2_angle;
56  if (!ft_solve(&eq))
57  return (false);
58  potential_hits[0] = eq.d1;
59  potential_hits[1] = eq.d2;
60  potential_hits[2] = ft_co_cap_hit(co, ray);
61  return (true);
