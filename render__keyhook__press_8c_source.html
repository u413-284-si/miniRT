59 int ft_keyhook_press(int key, t_render *render)
60 {
61  if (key == XK_Escape)
62  return (mlx_loop_end(render->mlx_ptrs.mlx_ptr));
63  else if (render->is_printing)
64  return (0);
65  else if (key == XK_i)
66  ft_toggle_menu(render);
67  else if (key == XK_Control_L)
68  ft_change_mode(render);
69  else if (key == XK_Shift_L)
70  ft_change_menu_page_ctrl(render);
71  else if (ft_is_manip_key(key))
72  {

74  ft_manip_scene(key, &render->scene, &render->active_hittable);
