39 bool ft_cy_calc_pot_hits(t_cylinder cy, t_ray ray, float potential_hits[4])
40 {
41  t_equation eq;
42  t_vec3 cap1_ray;
43  float dot_product_ray_axis;
44  float dot_product_cap_axis;
45 
46  cap1_ray = ft_vec3_sub(ray.origin, cy.cap1);
47  dot_product_ray_axis = ft_vec3_dot(ray.direction, cy.axis);
48  dot_product_cap_axis = ft_vec3_dot(cap1_ray, cy.axis);
49  eq.a = ft_vec3_dot(ray.direction, ray.direction) - \
50  dot_product_ray_axis * dot_product_ray_axis;
51  eq.b = 2.0 * (ft_vec3_dot(ray.direction, cap1_ray) - \
52  (dot_product_ray_axis * dot_product_cap_axis));
53  eq.c = ft_vec3_dot(cap1_ray, cap1_ray) - \
54  dot_product_cap_axis * dot_product_cap_axis - cy.r * cy.r;
55  if (ft_solve(&eq) < 0)
56  return (false);
57  potential_hits[0] = eq.d1;
58  potential_hits[1] = eq.d2;
59  potential_hits[2] = ft_cy_cap_hit(cy, ray, cy.cap1);
60  potential_hits[3] = ft_cy_cap_hit(cy, ray, cy.cap2);
61  return (true);
62 }
